# 作业2：股票可视化 Skill

## 功能概述

定义一个 Skill，用于对股票数据进行可视化分析，包括：
- 日波动和周波动绘制在同一张图中
- 基于波动幅度给出最佳买入/卖出时间建议

---

## 技术栈

| 组件 | 选择 | 说明 |
|------|------|------|
| 数据获取 | `yfinance` | Yahoo Finance API，免费获取股票数据 |
| 可视化 | `matplotlib` + `mplfinance` | 专业的金融图表库 |
| 数据处理 | `pandas` + `numpy` | 时间序列处理 |
| 技能封装 | Python Class | 可复用的 Skill 类 |

## 核心代码实现

### 1. 定义 StockAnalysisSkill 类

```python
import yfinance as yf
import pandas as pd
import numpy as np
import matplotlib.pyplot as plt
import mplfinance as mpf
from datetime import datetime, timedelta
from typing import Optional, Tuple, List

class StockAnalysisSkill:
    """
    股票分析与可视化 Skill
    
    功能：
    1. 获取股票历史数据
    2. 计算日波动和周波动
    3. 绘制组合图表（K线 + 日波动 + 周波动）
    4. 给出最佳买入/卖出时间建议
    """

    def __init__(self, ticker: str, period: str = "3mo"):
        """
        初始化 StockAnalysisSkill
        
        Args:
            ticker: 股票代码，如 'AAPL', 'MSFT', 'BABA'
            period: 时间范围，如 '1mo', '3mo', '6mo', '1y'
        """
        self.ticker = ticker
        self.period = period
        self.df: Optional[pd.DataFrame] = None
        self.daily_volatility: Optional[pd.Series] = None
        self.weekly_volatility: Optional[pd.Series] = None

    def fetch_data(self) -> pd.DataFrame:
        """
        从 Yahoo Finance 获取股票数据
        """
        print(f"正在获取 {self.ticker} 的股票数据（{self.period}）...")
        stock = yf.Ticker(self.ticker)
        self.df = stock.history(period=self.period)
        
        if self.df.empty:
            raise ValueError(f"未获取到 {self.ticker} 的数据，请检查股票代码")
        
        print(f"获取到 {len(self.df)} 条数据记录")
        return self.df

    def calculate_volatility(self) -> Tuple[pd.Series, pd.Series]:
        """
        计算日波动和周波动
        
        日波动 = (当日最高价 - 当日最低价) / 当日开盘价 * 100
        周波动 = 当周收盘价相对于周开盘价的变化率 * 100
        """
        if self.df is None or self.df.empty:
            raise ValueError("请先获取股票数据")

        # 日波动率
        self.daily_volatility = (
            (self.df["High"] - self.df["Low"]) / self.df["Open"] * 100
        )

        # 周波动率（按周重采样）
        weekly_df = self.df.resample("W").agg({
            "Open": "first",
            "High": "max",
            "Low": "min",
            "Close": "last",
            "Volume": "sum"
        })
        self.weekly_volatility = (
            (weekly_df["Close"] - weekly_df["Open"]) / weekly_df["Open"] * 100
        )

        return self.daily_volatility, self.weekly_volatility

    def plot_chart(self, save_path: Optional[str] = None) -> None:
        """
        绘制组合图表：K线图 + 日波动 + 周波动
        
        Args:
            save_path: 图片保存路径，如果不指定则显示到屏幕
        """
        if self.df is None:
            raise ValueError("请先获取股票数据")
        if self.daily_volatility is None:
            self.calculate_volatility()

        # 创建 3 个子图
        fig, axes = plt.subplots(3, 1, figsize=(14, 10), gridspec_kw={"height_ratios": [3, 1, 1]})
        fig.suptitle(f"{self.ticker} 股票分析（{self.period}）", fontsize=16, fontweight="bold")

        # 子图1：K线图
        mpf.plot(
            self.df,
            type="candle",
            style="charles",
            title=f"\n{self.ticker} K线图",
            ylabel="价格 ($)",
            ax=axes[0],
            show_nontrading=False,
        )

        # 子图2：日波动
        axes[1].plot(
            self.daily_volatility.index,
            self.daily_volatility.values,
            color="orange",
            linewidth=1,
            label="日波动率"
        )
        axes[1].axhline(
            y=self.daily_volatility.mean(),
            color="red",
            linestyle="--",
            label=f"平均日波动: {self.daily_volatility.mean():.2f}%"
        )
        axes[1].set_ylabel("日波动 (%)")
        axes[1].legend(loc="upper right")
        axes[1].grid(True, alpha=0.3)

        # 子图3：周波动
        axes[2].bar(
            self.weekly_volatility.index,
            self.weekly_volatility.values,
            color=[
                "green" if v >= 0 else "red"
                for v in self.weekly_volatility.values
            ],
            alpha=0.7,
            label="周波动率"
        )
        axes[2].axhline(
            y=0,
            color="black",
            linewidth=0.8
        )
        axes[2].set_ylabel("周波动 (%)")
        axes[2].legend(loc="upper right")
        axes[2].grid(True, alpha=0.3)
        axes[2].tick_params(axis="x", rotation=45)

        plt.tight_layout()

        if save_path:
            plt.savefig(save_path, dpi=150, bbox_inches="tight")
            print(f"图表已保存到: {save_path}")
        else:
            plt.show()

    def analyze_trading_signals(self) -> dict:
        """
        分析最佳买入/卖出时间建议
        
        基于波动幅度进行分析：
        - 最佳买入时间：周波动跌幅最大时（低位买入）
        - 最佳卖出时间：周波动涨幅最大时（高位卖出）
        
        Returns:
            包含分析结果的字典
        """
        if self.weekly_volatility is None:
            self.calculate_volatility()

        # 找出周波动最低点（最佳买入时机）
        min_week_idx = self.weekly_volatility.idxmin()
        min_week_val = self.weekly_volatility.min()
        min_week_date = min_week_idx.strftime("%Y-%m-%d")

        # 找出周波动最高点（最佳卖出时机）
        max_week_idx = self.weekly_volatility.idxmax()
        max_week_val = self.weekly_volatility.max()
        max_week_date = max_week_idx.strftime("%Y-%m-%d")

        # 计算平均波动
        avg_daily = self.daily_volatility.mean()
        avg_weekly = self.weekly_volatility.mean()

        # 计算波动标准差（衡量风险）
        std_daily = self.daily_volatility.std()
        std_weekly = self.weekly_volatility.std()

        # 获取对应日期的价格信息
        buy_price = self.df.loc[min_week_idx, "Close"] if min_week_idx in self.df.index else None
        sell_price = self.df.loc[max_week_idx, "Close"] if max_week_idx in self.df.index else None

        result = {
            "股票代码": self.ticker,
            "分析周期": self.period,
            "最佳买入时间": min_week_date,
            "当周跌幅": f"{min_week_val:.2f}%",
            "建议买入价格": f"${buy_price:.2f}" if buy_price else "N/A",
            "最佳卖出时间": max_week_date,
            "当周涨幅": f"{max_week_val:.2f}%",
            "建议卖出价格": f"${sell_price:.2f}" if sell_price else "N/A",
            "平均日波动": f"{avg_daily:.2f}%",
            "平均周波动": f"{avg_weekly:.2f}%",
            "日波动标准差": f"{std_daily:.2f}%",
            "周波动标准差": f"{std_weekly:.2f}%",
            "风险等级": "高" if std_weekly > 5 else "中" if std_weekly > 2 else "低",
        }

        return result

    def print_analysis_report(self) -> None:
        """
        打印完整的分析报告
        """
        signals = self.analyze_trading_signals()
        
        print("\n" + "=" * 50)
        print(f"股票分析报告: {signals['股票代码']}")
        print("=" * 50)
        print(f"分析周期: {signals['分析周期']}")
        print("-" * 50)
        print("【交易建议】")
        print(f"  最佳买入时间: {signals['最佳买入时间']}")
        print(f"  当周跌幅: {signals['当周跌幅']}")
        print(f"  建议买入价格: {signals['建议买入价格']}")
        print(f"  最佳卖出时间: {signals['最佳卖出时间']}")
        print(f"  当周涨幅: {signals['当周涨幅']}")
        print(f"  建议卖出价格: {signals['建议卖出价格']}")
        print("-" * 50)
        print("【波动性分析】")
        print(f"  平均日波动: {signals['平均日波动']}")
        print(f"  平均周波动: {signals['平均周波动']}")
        print(f"  日波动标准差: {signals['日波动标准差']}")
        print(f"  周波动标准差: {signals['周波动标准差']}")
        print(f"  风险等级: {signals['风险等级']}")
        print("=" * 50 + "\n")


# ==========================
# 使用示例
# ==========================
def main():
    """
    示例：分析苹果公司（AAPL）股票
    """
    # 初始化 Skill
    skill = StockAnalysisSkill(ticker="AAPL", period="3mo")

    # 获取数据
    skill.fetch_data()

    # 计算波动
    skill.calculate_volatility()

    # 绘制图表
    skill.plot_chart(save_path="aapl_analysis.png")

    # 打印分析报告
    skill.print_analysis_report()


if __name__ == "__main__":
    main()
```

## 注意事项

1. **网络依赖**：需要从 Yahoo Finance 获取数据，请确保网络畅通
2. **股票代码**：美股代码直接使用 ticker（如 AAPL），A股需要使用完整代码（如 600519.SS）
3. **matplotlib 中文显示**：如果出现中文乱码，需要设置字体：
   ```python
   plt.rcParams["font.sans-serif"] = ["SimHei"]  # Windows
   plt.rcParams["axes.unicode_minus"] = False
   ```
4. **数据延迟**：Yahoo Finance 数据可能有 15 分钟延迟，不适合实时交易
